🚀 Quick Start

# Clone the repository
git clone https://github.com/gourav53/bytebuzz-chat-app.git             
cd bytebuzz-chat-app

# Run backend
cd bytebuzz-backend
mvn spring-boot:run

# Run frontend
cd ../bytebuzz-frontend
npm install
npm run dev
